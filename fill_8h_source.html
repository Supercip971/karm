131 always_inline f64 transform(Math::Vec2f pos) const {
132 pos = pos - _start;
133 pos = pos.rotate(-(_end - _start).angle());
134 f64 scale = (_end - _start).len();
135 pos = pos / scale;
136
137 switch (_type) {
138 case LINEAR:

144 case CONICAL:
