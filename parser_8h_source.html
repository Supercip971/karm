62 static Res<Cmap::Table> chooseCmap(Parser& font) {
63 Opt<Cmap::Table> bestCmap = NONE;
64 isize bestScore = 0;
65
66 struct KnowCmap {
67 isize platformId;
68 isize encodingId;
69 isize type;
70 isize score;
71 };
72
73 Array<KnowCmap, 4> knowCmaps = {
74 KnowCmap{0, 3, 4, 150},
75 KnowCmap{3, 1, 4, 100},
76 KnowCmap{0, 4, 12, 1050},
77 KnowCmap{3, 10, 12, 1000},
78 };
79
80 for (auto table : font._cmap.iterTables()) {
81 for (auto& knowCmap : knowCmaps) {
82 if (knowCmap.platformId == table.platformId and
83 knowCmap.encodingId == table.encodingId and
84 knowCmap.type == table.type and
85 knowCmap.score > bestScore) {
86
87 bestCmap = table;
88 bestScore = knowCmap.score;
89 }
90 }
91 }
92
93 if (not bestCmap) {
94 logError("ttf: no suitable cmap table found");
95 return Error::other("no suitable cmap table found");
96 }
97
98 return Ok(*bestCmap);
99 }

106 static Res<Parser> init(Bytes slice) {
107 Parser font{slice};
108
109 if (font.version() != 0x00010000 and
110 font.version() != 0x4F54544F) {
111 logError("ttf: version {x} is not supported", (u64)font.version());
112 return Error::other("invalid version");
113 }
114
115 font._head = try$(font.requireTable<Head>());
116 font._cmap = try$(font.requireTable<Cmap>());
117 font._cmapTable = try$(chooseCmap(font));
118 font._glyf = try$(font.requireTable<Glyf>());
119 font._loca = try$(font.requireTable<Loca>());
120 font._hhea = try$(font.requireTable<Hhea>());
121 font._hmtx = try$(font.requireTable<Hmtx>());
122 font._gpos = font.lookupTable<Gpos>();
123 font._gsub = font.lookupTable<Gsub>();
124 font._name = font.lookupTable<Name>();
125 font._post = font.lookupTable<Post>();
126 if (not font._post.present())
127 logWarn("ttf: 'post' table not found");
128
129 font._os2 = font.lookupTable<Os2>();
130 if (not font._os2.present())
131 logWarn("ttf: 'OS/2' table not found");
132
133 return Ok(font);
134 }

148 auto iterTables() {
149 auto scan = begin();
150 /* auto version = */ scan.nextU32be();
151 auto numTables = scan.nextU16be();
152
153 /* auto searchRange = */ scan.nextU16be();
154 /* auto entrySelector = */ scan.nextU16be();
155 /* auto rangeShift = */ scan.nextU16be();
156
157 struct Table {
158 Str tag;
159 u32 checkSum;
160 u32 offset;
161 u32 length;
162 };
163
164 return Iter{[scan, i = 0uz, numTables] mutable -> Opt<Table> {
165 if (i == numTables) {
166 return NONE;
167 }
168
169 Table table{};
170 table.tag = scan.nextStr(4);
171 table.checkSum = scan.nextU32be();
172 table.offset = scan.nextU32be();
173 table.length = scan.nextU32be();
174
175 i++;
176 return table;
177 }};
178 }

180 template <typename T>
181 T lookupTable() {
182 for (auto table : iterTables()) {
183 if (table.tag == T::SIG) {
184 return T{sub(_slice, table.offset, table.offset + table.length)};
185 }
186 }
187
188 logDebugIf(DEBUG_PARSER, "ttf: '{}' table not found", T::SIG);
189 return T{};
190 }
191
192 template <typename T>
193 Res<T> requireTable() {
194 for (auto table : iterTables()) {
195 if (table.tag == T::SIG) {
196 return Ok(T{sub(_slice, table.offset, table.offset + table.length)});
197 }
198 }
199
200 logError("ttf: '{}' table not found", T::SIG);
201 return Error::other("table not found");
202 }
