122Res<> encodeName(Io::BEmit& e, Str name) {
123 for (auto part : iterSplit(name, '.')) {

126 }
127 e.writeU8be(0);

131Res<usize> decodeName(Cursor<u8> const start, Cursor<u8> curr, StringBuilder& out) {
132 usize len = 0;
133 while (not curr.ended()) {
134 auto b = curr.next();
135 len++;
136
137 if (b == 0) {
138 // End of name
139 if (out.len() > MAX_NAME_LEN)
140 return Error::invalidData("dns name too long");
141
142 return Ok(len);
143 } else if ((b & 0xc0) == 0xc0) {
144 // Pointer
145 if (curr.ended())
146 return Error::unexpectedEof("unexpected end of dns name");
147 usize off = ((b & 0x3f) << 8) | curr.next();
148 len++;
149 if (off >= start.len())
150 return Error::invalidData("invalid dns name pointer");
151 auto next = start;
152 next.next(off);
153 try$(decodeName(start, next, out));
154 return Ok(len);
155 } else {
156 // Label
157 if (b > MAX_LABEL_LEN)
158 return Error::invalidData("dns label too long");
159 if (out.len() + b + 1 > MAX_NAME_LEN)
160 return Error::invalidData("dns name too long");
161 if (out.len() > 0)
162 out.append('.');
163 out.append(curr.next(b).cast<char>());
164 len += b;
165 }
166 }
167
168 return Error::unexpectedEof("unexpected end of dns name");
169}

194 static Res<Buf<u8>> encode(Packet const& p) {
195 Io::BufferWriter buf;
196 Io::BEmit e(buf);
197
198 e.writeBytes(p.header().bytes());
199
200 for (auto& q : p._qs) {
201 try$(encodeName(e, q.name));
202 e.writeU16be(q.type);
203 e.writeU16be(q.class_);
204 }
205
206 for (auto& a : p._ans) {
207 try$(encodeName(e, a.name));
208 e.writeU16be(a.type);
209 e.writeU16be(a.class_);
210 e.writeU32be(a.ttl.toSecs());
211 e.writeU16be(a.data.len());
212 e.writeBytes(a.data);
213 }
214
215 return Ok(buf.take());
216 }
217
218 static Res<Packet> decode(Bytes buf) {
219 Io::BScan s{buf};
220
221 if (s.rem() < sizeof(Header))
222 return Error::unexpectedEof("packet too short");
223
224 Header hdr;
225 s.readTo(&hdr);
226
227 Packet pkt{hdr.id, (Flags)hdr.flags.value()};
228 Vec<Question>& qs = pkt._qs;
229 for (auto i = 0; i < hdr.qdcount; i++) {
230 StringBuilder name;
231 s.skip(try$(decodeName(buf, s.remBytes(), name)));
232
233 Question q{
234 name.take(),
235 (Type)s.nextU16be(),
236 (Class)s.nextU16be(),
237 };
238
239 qs.pushBack(std::move(q));
240 }
241
242 Vec<Answer>& ans = pkt._ans;
243 for (auto i = 0; i < hdr.ancount; i++) {
244 StringBuilder name;
245 s.skip(try$(decodeName(buf, s.remBytes(), name)));
246
247 Answer a;
248 a.name = name.take();
249 a.type = (Type)s.nextU16be();
250 a.class_ = (Class)s.nextU16be();
251 a.ttl = Duration::fromSecs(s.nextU32be());
252 auto len = s.nextU16be();
253 a.data = s.nextBytes(len);
254
255 ans.pushBack(std::move(a));
256 }
257
258 return Ok(std::move(pkt));
259 }

275 Res<Vec<Answer>> ask(Slice<Question> qs) {
276 Packet req{420, Flags::RD};
277 req._qs = qs;
278
279 logDebug("sending dns request to {}", _server);
280 try$(_conn.send(try$(Packet::encode(req)), _server));
281
282 logDebug("waiting for dns response");
283 Array<u8, 4096> buf;
284 auto [len, addr] = try$(_conn.recv(buf));
285
286 logDebug("received dns response from {}", addr);
287 if (addr != _server)
288 return Error::invalidData("received response from wrong address");
289
290 Packet resp = try$(Packet::decode(sub(buf, 0, len)));
291
292 if (resp._id != req._id)
293 return Error::invalidData("received response with wrong id");
294
295 if (resp.header().rcode() != RCode::NO_ERROR)
296 return Error::invalidData("received response with error code");
297
298 return Ok(std::move(resp._ans));
299 }

301 Res<Sys::Ip> resolve(Str host) {
302 auto qs = Vec<Question>{};
303 qs.pushBack(Question{host, Type::A, Class::IN});
304
305 auto answers = try$(ask(qs));
306 if (answers.len() == 0)
307 return Error::invalidData("no answers");
308
309 auto ans = answers[0];
310 if (ans.type != Type::A)
311 return Error::invalidData("invalid answer type");
312
313 if (ans.data.len() != 4)
314 return Error::invalidData("invalid answer data");
315
316 Sys::Ip4 ip{
317 ans.data[0],
318 ans.data[1],
319 ans.data[2],
320 ans.data[3],
321 };
322
323 return Ok(ip);
