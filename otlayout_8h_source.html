127struct CoverageTable : Io::BChunk {
128 using Format = Io::BField<u16be, 0>;
129 using GlyphCount = Io::BField<u16be, 2>;
130
131 usize format() const { return get<Format>(); }
132
133 usize len() const { return get<GlyphCount>(); }
134
135 Opt<usize> coverageIndex(usize glyphId) {
136 auto s = begin().skip(4);
137
138 if (format() == 1) {
139 for (auto i : range(len())) {
140 auto glyph = s.nextU16be();
141 if (glyph == glyphId) {
142 return i;
143 }
144 }
145 }
146
147 if (format() == 2) {
148 for (auto i : range(len())) {
149 (void)i;
150 auto start = s.nextU16be();
151 auto end = s.nextU16be();
152 auto index = s.nextU16be();
153 if (start <= glyphId and glyphId <= end) {
154 return index + glyphId - start;
155 }
156 }
157 }
158
159 return NONE;
160 }

225struct GlyphPairAdjustment : LookupSubtableBase {
226 static constexpr int FORMAT = 1;
227
228 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
229 auto s = begin();
230
231 // Read the table header
232 /* format = */ s.nextU16be();
233 auto coverageOffset = s.nextU16be();
234 auto valueFormat1 = s.nextU16be();
235 auto valueFormat2 = s.nextU16be();
236 auto pairSetCount = s.nextU16be();
237
238 // Lookup the coverage index for the first glyph
239 CoverageTable coverage{begin().skip(coverageOffset).remBytes()};
240 auto coverageIndex = coverage.coverageIndex(prev);
241
242 if (not coverageIndex or (*coverageIndex >= pairSetCount))
243 return NONE;
244
245 auto value1len = ValueRecord::len(valueFormat1);
246 auto value2len = ValueRecord::len(valueFormat2);
247 auto pairSetOffset = s.skip(coverageIndex.unwrap() * 2).nextU16be();
248
249 // Lookup the PairSet table for the second glyph
250 auto pairSetTable = begin().skip(pairSetOffset);
251 auto pairValueCount = pairSetTable.nextU16be();
252
253 for (usize i : range(pairValueCount)) {
254 (void)i;
255 auto secondGlyph = pairSetTable.nextU16be();
256 if (secondGlyph == curr) {
257 ValueRecord value1 = ValueRecord::read(pairSetTable, valueFormat1);
258 ValueRecord value2 = ValueRecord::read(pairSetTable, valueFormat2);
259 return Pair<ValueRecord>{value1, value2};
260 }
261
262 pairSetTable.skip(value1len + value2len);
263 }
264
265 return NONE;
266 }

270struct ClassDef : Io::BChunk {
271 Opt<usize> classOf(usize glyphId) {
272 auto s = begin();
273 auto format = s.nextU16be();
274
275 if (format == 1) {
276 auto startGlyph = s.nextU16be();
277 auto glyphCount = s.nextU16be();
278 if (startGlyph <= glyphId and glyphId < startGlyph + glyphCount) {
279 return s.skip((glyphId - startGlyph) * 2).nextU16be();
280 }
281 }
282
283 if (format == 2) {
284 auto classRangeCount = s.nextU16be();
285 for (usize i : range(classRangeCount)) {
286 (void)i;
287 auto startGlyph = s.nextU16be();
288 auto endGlyph = s.nextU16be();
289 auto glyphClass = s.nextU16be();
290 if (startGlyph <= glyphId and glyphId <= endGlyph) {
291 return glyphClass;
292 }
293 }
294 }
295
296 return NONE;
297 }

301struct ClassPairAdjustment : LookupSubtableBase {
302 static constexpr int FORMAT = 2;
303
304 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
305 auto s = begin();
306
307 // Read the table header
308 /* format = */ s.nextU16be();
309 /* coverageOffset = */ s.nextU16be();
310 auto valueFormat1 = s.nextU16be();
311 auto valueFormat2 = s.nextU16be();
312 auto classDef1Offset = s.nextU16be();
313 auto classDef2Offset = s.nextU16be();
314 /* class1Count = */ s.nextU16be();
315 auto class2Count = s.nextU16be();
316
317 ClassDef prevClassDef{begin().skip(classDef1Offset).remBytes()};
318 ClassDef currClassDef{begin().skip(classDef2Offset).remBytes()};
319
320 auto prevClass = try$(prevClassDef.classOf(prev));
321 auto currClass = try$(currClassDef.classOf(curr));
322
323 auto value1len = ValueRecord::len(valueFormat1);
324 auto value2len = ValueRecord::len(valueFormat2);
325
326 auto class2Size = value1len + value2len;
327 auto class1Size = class2Count * class2Size;
328 s.skip((prevClass * class1Size) + (currClass * class2Size));
329
330 ValueRecord value1 = ValueRecord::read(s, valueFormat1);
331 ValueRecord value2 = ValueRecord::read(s, valueFormat2);
332
333 return Pair<ValueRecord>{value1, value2};
334 }
