128struct CoverageTable : Io::BChunk {
129 using Format = Io::BField<u16be, 0>;
130 using GlyphCount = Io::BField<u16be, 2>;
131
132 usize format() const { return get<Format>(); }
133
134 usize len() const { return get<GlyphCount>(); }
135
136 Opt<usize> coverageIndex(usize glyphId) {
137 auto s = begin().skip(4);
138
139 if (format() == 1) {
140 for (auto i : range(len())) {
141 auto glyph = s.nextU16be();
142 if (glyph == glyphId) {
143 return i;
144 }
145 }
146 }
147
148 if (format() == 2) {
149 for (auto i : range(len())) {
150 (void)i;
151 auto start = s.nextU16be();
152 auto end = s.nextU16be();
153 auto index = s.nextU16be();
154 if (start <= glyphId and glyphId <= end) {
155 return index + glyphId - start;
156 }
157 }
158 }
159
160 return NONE;
161 }

226struct GlyphPairAdjustment : LookupSubtableBase {
227 static constexpr int FORMAT = 1;
228
229 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
230 auto s = begin();
231
232 // Read the table header
233 /* format = */ s.nextU16be();
234 auto coverageOffset = s.nextU16be();
235 auto valueFormat1 = s.nextU16be();
236 auto valueFormat2 = s.nextU16be();
237 auto pairSetCount = s.nextU16be();
238
239 // Lookup the coverage index for the first glyph
240 CoverageTable coverage{begin().skip(coverageOffset).remBytes()};
241 auto coverageIndex = coverage.coverageIndex(prev);
242
243 if (not coverageIndex or (*coverageIndex >= pairSetCount))
244 return NONE;
245
246 auto value1len = ValueRecord::len(valueFormat1);
247 auto value2len = ValueRecord::len(valueFormat2);
248 auto pairSetOffset = s.skip(coverageIndex.unwrap() * 2).nextU16be();
249
250 // Lookup the PairSet table for the second glyph
251 auto pairSetTable = begin().skip(pairSetOffset);
252 auto pairValueCount = pairSetTable.nextU16be();
253
254 for (usize i : range(pairValueCount)) {
255 (void)i;
256 auto secondGlyph = pairSetTable.nextU16be();
257 if (secondGlyph == curr) {
258 ValueRecord value1 = ValueRecord::read(pairSetTable, valueFormat1);
259 ValueRecord value2 = ValueRecord::read(pairSetTable, valueFormat2);
260 return Pair<ValueRecord>{value1, value2};
261 }
262
263 pairSetTable.skip(value1len + value2len);
264 }
265
266 return NONE;
267 }

271struct ClassDef : Io::BChunk {
272 Opt<usize> classOf(usize glyphId) {
273 auto s = begin();
274 auto format = s.nextU16be();
275
276 if (format == 1) {
277 auto startGlyph = s.nextU16be();
278 auto glyphCount = s.nextU16be();
279 if (startGlyph <= glyphId and glyphId < startGlyph + glyphCount) {
280 return s.skip((glyphId - startGlyph) * 2).nextU16be();
281 }
282 }
283
284 if (format == 2) {
285 auto classRangeCount = s.nextU16be();
286 for (usize i : range(classRangeCount)) {
287 (void)i;
288 auto startGlyph = s.nextU16be();
289 auto endGlyph = s.nextU16be();
290 auto glyphClass = s.nextU16be();
291 if (startGlyph <= glyphId and glyphId <= endGlyph) {
292 return glyphClass;
293 }
294 }
295 }
296
297 return NONE;

302struct ClassPairAdjustment : LookupSubtableBase {
303 static constexpr int FORMAT = 2;
304
305 Opt<Pair<ValueRecord>> adjustments(usize prev, usize curr) {
306 auto s = begin();
307
308 // Read the table header
309 /* format = */ s.nextU16be();
310 /* coverageOffset = */ s.nextU16be();
311 auto valueFormat1 = s.nextU16be();
312 auto valueFormat2 = s.nextU16be();
313 auto classDef1Offset = s.nextU16be();
314 auto classDef2Offset = s.nextU16be();
315 /* class1Count = */ s.nextU16be();
316 auto class2Count = s.nextU16be();
317
318 ClassDef prevClassDef{begin().skip(classDef1Offset).remBytes()};
319 ClassDef currClassDef{begin().skip(classDef2Offset).remBytes()};
320
321 auto prevClass = try$(prevClassDef.classOf(prev));
322 auto currClass = try$(currClassDef.classOf(curr));
323
324 auto value1len = ValueRecord::len(valueFormat1);
325 auto value2len = ValueRecord::len(valueFormat2);
326
327 auto class2Size = value1len + value2len;
328 auto class1Size = class2Count * class2Size;
329 s.skip((prevClass * class1Size) + (currClass * class2Size));
330
331 ValueRecord value1 = ValueRecord::read(s, valueFormat1);
332 ValueRecord value2 = ValueRecord::read(s, valueFormat2);
333
334 return Pair<ValueRecord>{value1, value2};
335 }
