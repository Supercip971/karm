103inline constexpr Expr auto nOrN(usize atLeast, usize atMost, Expr auto expr) {
104 return [=](auto& scan) {
105 usize count = 0;
106 auto saved = scan;
107 saved = scan;
108 while (expr(scan)) {
109 count++;
110 saved = scan;
111 if (count == atMost)
112 break;
113 }
114 if (count >= atLeast) {
115 scan = saved;
116 return true;
117 }
118 return false;
119 };
