162template <Meta::Enum E>
163consteval Meta::UnderlyingType<E> enumMin() {
164 if constexpr (requires { E::_MIN; }) {
165 return toUnderlyingType(E::_MIN);
166 }
167#define ITER(LEN) \
168 else if constexpr (requires { E::_LEN##LEN; }) { \
169 return Meta::UnderlyingType<E>{}; \

194template <Meta::Enum E>
195consteval Meta::UnderlyingType<E> enumLen() {
196#define ITER(LEN) \
