704template <StaticEncoding E>
705struct StringFormatter {
706 bool prefix = false;
707
708 void parse(Io::SScan& scan) {
709 if (scan.skip('#'))
710 prefix = true;
711 }
712
713 Res<> format(Io::TextWriter& writer, _Str<E> text) {
714 if (not prefix)
715 return writer.writeStr(text);
716
717 try$(writer.writeRune('"'));
718 for (Rune c : iterRunes(text)) {
719 if (c == '"')
720 try$(writer.writeStr("\\\""s));
721 else if (c == '\\')
722 try$(writer.writeStr("\\\\"s));
723 else if (c == '\a')
724 try$(writer.writeStr("\\a"s));
725 else if (c == '\b')
726 try$(writer.writeStr("\\b"s));
727 else if (c == '\f')
728 try$(writer.writeStr("\\f"s));
729 else if (c == '\n')
730 try$(writer.writeStr("\\n"s));
731 else if (c == '\r')
732 try$(writer.writeStr("\\r"s));
733 else if (c == '\t')
734 try$(writer.writeStr("\\t"s));
735 else if (c == '\v')
736 try$(writer.writeStr("\\v"s));
737 else if (not isAsciiPrint(c))
738 try$(Io::format(writer, "\\u{x}", c));
739 else
740 try$(writer.writeRune(c));
741 }
742 try$(writer.writeRune('"'));

744 }
