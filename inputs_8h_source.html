38static inline bool match(KeyMod in, KeyMod mods) {
39 KeyMod either = KeyMod::NONE;
40 KeyMod mask = KeyMod::NONE;
41
42 // Removing the non-combinable modifiers from the matching
43 // Because eg: NumLock + A should be considered exactly like A
44 if (static_cast<bool>(in & KeyMod::NUM)) {
45 mask |= KeyMod::NUM;
46 }
47 if (static_cast<bool>(in & KeyMod::CAPS)) {
48 mask |= KeyMod::CAPS;
49 }
50 if (static_cast<bool>(in & KeyMod::MODE)) {
51 mask |= KeyMod::MODE;
52 }
53 if (static_cast<bool>(in & KeyMod::SCROLL)) {
54 mask |= KeyMod::SCROLL;
55 }
56
57 if (static_cast<bool>(in & KeyMod::LSHIFT) or
58 static_cast<bool>(in & KeyMod::RSHIFT)) {
59 either |= KeyMod::SHIFT;
60 mask |= KeyMod::LSHIFT | KeyMod::RSHIFT;
61 }
62
63 if (static_cast<bool>(in & KeyMod::LCTRL) or
64 static_cast<bool>(in & KeyMod::RCTRL)) {
65 either |= KeyMod::CTRL;
66 mask |= KeyMod::LCTRL | KeyMod::RCTRL;
67 }
68
69 if (static_cast<bool>(in & KeyMod::LALT) or
70 static_cast<bool>(in & KeyMod::RALT)) {
71 either |= KeyMod::ALT;
72 mask |= KeyMod::LALT | KeyMod::RALT;
73 }
74
75 if (static_cast<bool>(in & KeyMod::LSUPER) or
76 static_cast<bool>(in & KeyMod::RSUPER)) {
77 either |= KeyMod::SUPER;
78 mask |= KeyMod::LSUPER | KeyMod::RSUPER;
79 }
80
81 return (((in | either) & mods) == mods) and
82 ((in & (mods | mask)) == in);
83}
