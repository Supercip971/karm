248 static Time fromDuration(Duration span) {
249 Time result;
250 result.hour = span.toHours();
251 span -= Duration::fromHours(result.hour);
252 result.minute = span.toMinutes();
253 span -= Duration::fromMinutes(result.minute);

461 static Date epoch() {

464
465 static Date fromInstant(SystemTime stamp) {
466 auto span = stamp - SystemTime::epoch();
467 auto days = span.toDays();
468
469 Year year{1970};
470 while (days >= usize(year.days())) {
471 days -= year.days();
472 year = year.next();
473 }
474
475 Month month{0};
476 while (days >= usize(year.daysIn(month))) {
477 days -= year.daysIn(month);
478 month = month.next();
479 }
480
481 return Date{days, month, year};
482 }
483
484 SystemTime toInstant() const {
485 usize days{};
486
487 for (isize y = 1970; y < isize(year); y++) {
488 days += Year{y}.isLeap() ? 366 : 365;
489 }
490
491 for (isize m = 0; m < isize(month); m++) {
492 days += year.daysIn(m);
493 }
494
495 days += day;
496
497 return SystemTime::epoch() + Duration::fromDays(days);
498 }
499
501 u8 dayOfMonth() const {
502 Date firstDay{0, month, year};
503 return (toInstant() - firstDay.toInstant()).toDays();
504 }
505
506 DayOfWeek dayOfWeek() const {
507 Date firstJan{0, 0, year};
508 return DayOfWeek((toInstant() - firstJan.toInstant()).toDays() % 7);
509 }
510
511 usize dayOfYear() const {
512 Date firstJan{0, 0, year};
513 return (toInstant() - firstJan.toInstant()).toDays();
514 }
