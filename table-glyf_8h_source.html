97 for (isize c = 0; c < m.numContours; c++) {

117
118 isize y = (flags & Y_SHORT_VECTOR)
119 ? ((flags & SAME_OR_POSITIVE_Y) ? yCoordsScan.nextU8be() : -yCoordsScan.nextU8be())

138 g.quadTo(cp, p1);
