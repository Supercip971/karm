29 _z = z;
30 _d = _w * sqrt(abs(z * z - 1));

41 if (not _init) {

43 _y = x;
44 _yd = 0;
45 }
46
47 // Estimate velocity
48 auto xd = (x - _xp) / t;
49 _xp = x;
50
51 // Clamp k1 and k2 to guarantee stability without jitter
52 f64 k1Stable;
53 f64 k2Stable;
54
55 if (_w * t < _z) {
56 k1Stable = _k1;
57 k2Stable = max(_k2, t * t / 2 + t * _k1 / 2, t * _k1);
58 } else {
59 // Use pole matching when the system is very fast
60 f64 t1 = exp(-_z * _w * t);
61 f64 alpha = 2 * t1 * (_z <= 1 ? cos(t * _d) : cosh(t * _d));
62 f64 beta = t1 * t1;
63 f64 t2 = t / (1 + beta - alpha);
64 k1Stable = (1 - beta) * t2;
65 k2Stable = t * t2;
66 }
67
68 // Integrate position by velocity
69 _y = _y + t * _yd;
70
71 // Interfeare velocity by acceleration
72 _yd = _yd + t * (x + _k3 * xd - _y - k1Stable * _yd) / k2Stable;
73
74 return _y;
75 }
