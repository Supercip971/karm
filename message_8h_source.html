101 };
102
103 Array<u8, CAP> _buf;
104 };

106 usize _len = 0;
107
108 Array<Handle, 16> _hnds;
109 usize _hndsLen = 0;

111 Header& header() {
112 return _header;
113 }

116 return _header;
117 }
118
119 usize len() const {
120 return _len;
121 }
122
123 Bytes bytes() {
124 return sub(_buf, 0, len());
125 }
126
127 Slice<Handle> handles() {

136 template <typename T, typename... Args>
137 static Res<Message> packReq(Port to, u64 seq, Args&&... args) {
138 T payload{std::forward<Args>(args)...};
139
140 Message msg;
141 msg._header = {
142 seq,
143 Port::INVALID,
144 to,
145 Meta::idOf<T>(),
146 };
147 Io::BufWriter reqBuf{msg._payload};
148 MessageWriter reqPack{reqBuf};
149
150 try$(Sys::pack(reqPack, payload));
151
152 msg._len = try$(Io::tell(reqBuf)) + sizeof(Header);
153
154 return Ok(std::move(msg));
155 }
156
157 template <typename T, typename... Args>
158 Res<Message> packResp(Args&&... args) {
159 typename T::Response payload{std::forward<Args>(args)...};
160
161 Message resp;
162 resp._header = {
163 header().seq,
164 header().to,
165 header().from,
166 Meta::idOf<typename T::Response>(),
167 };
168
169 Io::BufWriter respBuf{resp._payload};
170 MessageWriter respPack{respBuf};
171
172 try$(Sys::pack(respPack, payload));
173
174 resp._len = try$(Io::tell(respBuf)) + sizeof(Header);
175
176 return Ok(std::move(resp));
177 }

179 template <typename T>
180 Res<T> unpack() {
181 MessageReader s{bytes(), handles()};
182 if (not is<T>())
183 return Error::invalidData("unexpected message");
184 try$(Sys::unpack<Header>(s));

236struct MessagePacker<Union<Ts...>> {
237 static Res<> pack(MessageWriter& e, Union<Ts...> const& val) {
238 try$(Sys::pack<u8>(e, val.index()));
