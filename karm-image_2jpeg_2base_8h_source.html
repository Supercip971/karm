115struct BitWriter {
116 Io::BEmit& e;
117
118 u8 _buf = 0;
119 u8 _len = 0;
120
121 always_inline BitWriter(Io::BEmit& e) : e(e) {}
122
123 always_inline void writeBit(u8 bit) {
124 _buf |= (bit & 1) << (7 - _len);
125 _len = (_len + 1) % 8;
126
127 if (_len == 0) {
128 e.writeU8be(_buf);
129 if (_buf == 0xFF)
130 e.writeU8be(0);
131 _buf = 0;
132 }
133 }

136 for (usize i = 1; i <= len; ++i)

141struct BitReader {
142 Io::BScan& s;
143
144 u8 _buf = 0;
145 u8 _len = 0;
146
147 always_inline BitReader(Io::BScan& s) : s(s) {}
148
149 always_inline void reset() {
150 _buf = 0x0;
151 _len = 0x0;
152 }
153
154 always_inline Res<> fill() {
155 if (_len)
156 return Ok();
157
158 _buf = s.nextU8be();
159 while (_buf == 0xFF) {
160 u8 marker = s.peekU8be();
161
162 while (marker == 0xFF) {
163 s.nextU8be();
164 marker = s.peekU8be();
165 }
166
167 if (marker == 0x00) {
168 s.nextU8be();
169 break;
170 } else if (RST0 <= marker and marker <= RST7) {
171 s.nextU8be();
172 _buf = s.nextU8be();
173 } else {
174 logError("jpeg: invalid marker");
175 return Error::invalidData("invalid marker");
176 }
177 }
178
179 _len = 8;
180 return Ok();
181 }
182
183 always_inline Res<u8> nextBit() {
184 try$(fill());
185 u8 bit = _buf >> 7;
186 _buf <<= 1;
187 _len--;
188 return Ok(bit);
189 }
190
191 always_inline Res<usize> nextBits(u8 n) {
192 usize res = 0;
193 while (n--) {
194 res = (res << 1) | try$(nextBit());
195 }
196 return Ok(res);
197 }
198};
