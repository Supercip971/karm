456always_inline constexpr void sort(MutSliceable auto& slice, auto cmp) {
457 if (slice.len() <= 1)
458 return;
459
460 auto pivot = slice[slice.len() / 2];
461 auto left = 0uz;
462 auto right = slice.len() - 1;
463
464 while (left <= right) {
465 while (cmp(slice[left], pivot) < 0) {
466 left++;
467 }
468
469 while (cmp(slice[right], pivot) > 0) {
470 right--;
471 }
472
473 if (left <= right) {
474 std::swap(slice[left], slice[right]);
475 left++;
476 right--;
477 }
478 }
479
480 auto rightSlice = mutSub(slice, 0, right + 1);
481 sort(rightSlice, cmp);
482
483 auto leftSlice = mutSub(slice, left, slice.len());
484 sort(leftSlice, cmp);
