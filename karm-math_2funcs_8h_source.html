113template <Meta::Float T, Precision P = PRECISE>
114constexpr T cos(T x) {
115 constexpr T tp = 1. / (2. * PI);
116 x *= tp;
117 x -= T(.25) + floor(x + T(.25));
118 x *= T(16.) * (abs(x) - T(.5));
119 if constexpr (P == PRECISE) {
120 x += T(.225) * x * (abs(x) - T(1.));
121 }
122 return x;
123}

152template <Meta::Float T>
153T exp(T x) {
154 if (x == 0.0)
155 return 1;
156
157 if (isNegInf(x))
158 return 0;
159
160 if (isInf(x) == 1)
161 return x;
162
163 if (isNan(x))
164 return x;
165
166 if (x < 0.0)
167 return (1.0 / exp(-x));
168
169 T counter = 0;
170 T result = 1.0;
171 T term = 1.0;
172
173 while (++counter, x / counter > (x / (x * x + 42.0))) {
174 term *= x / counter;
175 result += term;
176 }
177
178 return (result);
179}

193template <Meta::Float T>
194constexpr T log2(T x) {
195 // References:
196 // Gist comparing some implementations
197 // * https://gist.github.com/Hendiadyoin1/f58346d66637deb9156ef360aa158bf9
198
199 if (x == 0)
200 return -INF;
201
202 if (x <= 0 or __builtin_isnan(x))
203 return NAN;
204
205 FloatBits<T> ext{.d = x};
206 T exponent = ext.exponent - FloatBits<T>::exponentBias;
207
208 // When the mantissa shows 0b00 (implicitly 1.0) we are on a power of 2
209 if (ext.mantissa == 0)
210 return exponent;
211
212 // FIXME: Handle denormalized numbers separately
213
214 // (1 <= mantissa < 2)
215 T m;
216 if constexpr (isLittleEndian()) {
217 m = ((FloatBits<T>){
218 .mantissa = ext.mantissa,
219 .exponent = FloatBits<T>::exponentBias,
220 .sign = ext.sign
221 })
222 .d;
223 } else {
224 m = ((FloatBits<T>){
225 .sign = ext.sign,
226 .exponent = FloatBits<T>::exponent_bias,
227 .mantissa = ext.mantissa
228 })
229 .d;
230 }
231
232 // This is a reconstruction of one of Sun's algorithms
233 // They use a transformation to lower the problem space,
234 // while keeping the precision, and a 14th degree polynomial,
235 // which is mirrored at sqrt(2)
236 // TODO: Sun has some more algorithms for this and other math functions,
237 // leveraging LUTs, investigate those, if they are better in performance and/or precision.
238 // These seem to be related to crLibM's implementations, for which papers and references
239 // are available.
240 // FIXME: Dynamically adjust the amount of precision between floats and doubles
241 // AKA floats might need less accuracy here, in comparison to doubles
242
243 bool inverted = false;
244 // m > sqrt(2)
245 if (m > SQRT2) {
246 inverted = true;
247 m = 2 / m;
248 }
249 T s = (m - 1) / (m + 1);
250 // ln((1 + s) / (1 - s)) == ln(m)
251 T s2 = s * s;
252 // clang-format off
253 T high_approx = s2 * (static_cast<T>(0.6666666666666735130)
254 + s2 * (static_cast<T>(0.3999999999940941908)
255 + s2 * (static_cast<T>(0.2857142874366239149)
256 + s2 * (static_cast<T>(0.2222219843214978396)
257 + s2 * (static_cast<T>(0.1818357216161805012)
258 + s2 * (static_cast<T>(0.1531383769920937332)
259 + s2 * static_cast<T>(0.1479819860511658591)))))));
260 // clang-format on
261
262 // ln(m) == 2 * s + s * high_approx
263 // log2(m) == log2(e) * ln(m)
264 T log2_mantissa = L2_E * (2 * s + s * high_approx);
265 if (inverted)
266 log2_mantissa = 1 - log2_mantissa;
267 return exponent + log2_mantissa;
268}

275template <Meta::Integral T = isize, Precision = PRECISE>
276constexpr T pow(T base, T exponent) {
277 if (exponent < 0)
278 return 0;
279 if (exponent == 0)
280 return 1;
281
282 T res = 1;
283 while (exponent > 0) {
284 if (exponent & 1)
285 res *= base;
286 base *= base;
287 exponent /= 2u;
288 }
289 return res;
290}

292template <Meta::Float T = f64, Precision = PRECISE>
293constexpr T pow(T base, T exponent) {
294 if (isNan(exponent))
295 return exponent;
296 if (exponent == 0)
297 return 1;
298 if (base == 0)
299 return 0;
300 if (exponent == 1)
301 return base;
302 int y_as_int = (int)exponent;
303 if (exponent == (T)y_as_int) {
304 T result = base;
305 for (int i = 0; i < abs(exponent) - 1; ++i)
306 result *= base;
307 if (exponent < 0)
308 result = 1.0l / result;
309 return result;
310 }
311
312 return exp2<T>(exponent * log2<T>(base));
313}

320template <Meta::Float T, Precision P = PRECISE>
321constexpr T asin(T x) {
322 if (x > 1 or x < -1)
323 return NAN;
324
325 if (x > (T)0.5 or x < (T)-0.5)
326 return 2 * atan<T>(x / (1 + sqrt<T>(1 - x * x)));
327
328 T squared = x * x;
329 T value = x;
330 T i = x * squared;
331
332 auto odd = [](this auto& self, auto i) {
333 if (i == 1)
334 return 1;
335 return i * self(i - 2);
336 };
337
338 auto even = [](this auto& self, auto i) {
339 if (i == 0)
340 return 2;
341 return i * self(i - 2);
342 };
343
344 for (auto it = 1; it < 15; it += 2) {
345 value += i * odd(it) / even(it + 1);
346 i *= squared;
347 }
348
349 return value;
350}
