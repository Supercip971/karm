229 static Family parse(String const& familyName) {

231 if (eqCi(familyNameView, "serif"s))
232 return GenericFamily::SERIF;

234 return GenericFamily::SANS_SERIF;

241 else if (eqCi(familyNameView, "system"s))
242 return GenericFamily::SYSTEM;

244 return GenericFamily::EMOJI;
245 else if (eqCi(familyNameView, "math"s))
246 return GenericFamily::MATH;
247 else if (eqCi(familyNameView, "fangsong"s))

282 auto operator<=>(FontAttrs const& other) const {
283 // NOTE: Comparaison order is important
284
285 auto ordr = family <=> other.family;
286 if (ordr != 0)
287 return ordr;

289 ordr = style <=> other.style;
290 if (ordr != 0)
291 return ordr;

293 ordr = stretch <=> other.stretch;
294 if (ordr != 0)
295 return ordr;
