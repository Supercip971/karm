92struct Utf8 {
93 using Unit = char;
94 using One = _Multiple<Unit, 4>;
95
96 always_inline static constexpr usize unitLen(Unit first) {
97 if ((first & 0xf8) == 0xf0)
98 return 4;
99 else if ((first & 0xf0) == 0xe0)
100 return 3;
101 else if ((first & 0xe0) == 0xc0)
102 return 2;
103 else
104 return 1;
105 }
106
107 always_inline static constexpr usize runeLen(Rune rune) {
108 if (rune <= 0x7f)
109 return 1;
110 else if (rune <= 0x7ff)
111 return 2;
112 else if (rune <= 0xffff)
113 return 3;
114 else
115 return 4;
116 }
117
118 always_inline static bool decodeUnit(Rune& result, DecodeInput<Unit> auto& in) {
119 if (in.rem() == 0) {
120 result = U'�';
121 return false;
122 }
123
124 Unit first = in.next();
125
126 if (unitLen(first) > in.rem() + 1) {
127 result = U'�';
128 return false;
129 }
130
131 if ((first & 0xf8) == 0xf0) {
132 result = ((0x07 & first) << 18) |
133 ((0x3f & in.next()) << 12) |
134 ((0x3f & in.next()) << 6) |
135 ((0x3f & in.next()));
136 } else if ((first & 0xf0) == 0xe0) {
137 result = ((0x0f & first) << 12) |
138 ((0x3f & in.next()) << 6) |
139 ((0x3f & in.next()));
140 } else if ((first & 0xe0) == 0xc0) {
141 result = ((0x1f & first) << 6) |
142 ((0x3f & in.next()));
143 } else {
144 result = first;
145 }
146
147 return true;
148 }
149
150 always_inline static bool encodeUnit(Rune c, EncodeOutput<Unit> auto& out) {
151 if (unitLen(c) > out.rem()) [[unlikely]]
152 panic("bad");
153
154 if (c <= 0x7f) {
155 out.put(c);
156 } else if (c <= 0x7ff) {
157 out.put(0xc0 | (c >> 6));
158 out.put(0x80 | (c & 0x3f));
159 } else if (c <= 0xffff) {
160 out.put(0xe0 | (c >> 12));
161 out.put(0x80 | ((c >> 6) & 0x3f));
162 out.put(0x80 | (c & 0x3f));
163 } else if (c <= 0x10ffff) {
164 out.put(0xf0 | (c >> 18));
165 out.put(0x80 | ((c >> 12) & 0x3f));
166 out.put(0x80 | ((c >> 6) & 0x3f));
167 out.put(0x80 | (c & 0x3f));
168 } else {
169 return encodeUnit(U'�', out);
170 }
171
172 return true;
173 }
174};

182struct Utf16 {
183 using Unit = u16;
184 using One = _Multiple<Unit, 2>;
185
186 always_inline static constexpr usize unitLen(Unit first) {
187 if (first >= 0xd800 and first <= 0xdbff)
188 return 2;
189 else
190 return 1;
191 }
192
193 always_inline static constexpr usize runeLen(Rune rune) {
194 if (rune <= 0xffff)
195 return 1;
196 else
197 return 2;
198 }
199
200 always_inline static bool decodeUnit(Rune& result, DecodeInput<Unit> auto& in) {
201 Unit first = in.next();
202
203 if (unitLen(first) > in.rem()) {
204 result = U'�';
205 return false;
206 }
207
208 if (first >= 0xd800 and first <= 0xdbff) {
209 if (in.rem() < 2) {
210 return false;
211 }
212
213 Unit second = in.next();
214
215 if (second < 0xdc00 or second > 0xdfff) {
216 return false;
217 }
218
219 result = ((first - 0xd800) << 10) | (second - 0xdc00) + 0x10000;
220 } else {
221 result = first;
222 }
223
224 return true;
225 }
226
227 always_inline static bool encodeUnit(Rune c, EncodeOutput<Unit> auto& out) {
228 if (c <= 0xffff) {
229 out.put(c);
230 return true;
231 } else if (c <= 0x10ffff) {
232 out.put(0xd800 | ((c - 0x10000) >> 10));
233 out.put(0xdc00 | ((c - 0x10000) & 0x3ff));
234 return true;
235 } else {
236 return encodeUnit(U'�', out);
237 }
238 }
