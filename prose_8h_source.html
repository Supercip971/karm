312 Lbc lbcAt(usize runeIndex) const {
313 auto maybeLi = searchLowerBound(
314 _lines, [&](Line const& l) {
315 return l.runeRange.start <=> runeIndex;
316 }
317 );
318
319 if (not maybeLi)
320 return {0, 0, 0};
321
322 auto li = *maybeLi;
323
324 auto& line = _lines[li];
325
326 auto maybeBi = searchLowerBound(
327 line.blocks(), [&](Block const& b) {
328 return b.runeRange.start <=> runeIndex;
329 }
330 );
331
332 if (not maybeBi)
333 return {li, 0, 0};
334
335 auto bi = *maybeBi;
336
337 auto& block = line.blocks()[bi];
338
339 auto maybeCi = searchLowerBound(
340 block.cells(), [&](Cell const& c) {
341 return c.runeRange.start <=> runeIndex;
342 }
343 );
344
345 if (not maybeCi)
346 return {li, bi, 0};
347
348 auto ci = *maybeCi;
349
350 auto cell = block.cells()[ci];
351
352 if (cell.runeRange.end() == runeIndex) {
353 // Handle the case where the rune is the last of the text
354 ci++;
355 }
356
357 return {li, bi, ci};
358 }
359
360 Vec2Au queryPosition(usize runeIndex) const {
361 auto [li, bi, ci] = lbcAt(runeIndex);
362
363 if (isEmpty(_lines))
364 return {};
365
366 auto& line = _lines[li];
367
368 if (isEmpty(line.blocks()))
369 return {0_au, line.baseline};
370
371 auto& block = line.blocks()[bi];
372
373 if (isEmpty(block.cells()))
374 return {block.pos, line.baseline};
375
376 if (ci >= block.cells().len()) {
377 // Handle the case where the rune is the last of the text
378 auto& cell = last(block.cells());
379 return {block.pos + cell.pos + cell.adv, cell.Yposition(line.baseline)};
380 }
381
382 auto& cell = block.cells()[ci];
383
384 return {block.pos + cell.pos, cell.Yposition(line.baseline)};
385 }
