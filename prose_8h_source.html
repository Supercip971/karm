343 struct Lbc {
344 usize li, bi, ci;
345 };
346
347 Lbc lbcAt(usize runeIndex) const {
348 auto maybeLi = searchLowerBound(
349 _lines, [&](Line const& l) {
350 return l.runeRange.start <=> runeIndex;
351 }
352 );
353
354 if (not maybeLi)
355 return {0, 0, 0};
356
357 auto li = *maybeLi;
358
359 auto& line = _lines[li];
360
361 auto maybeBi = searchLowerBound(
362 line.blocks(), [&](Block const& b) {
363 return b.runeRange.start <=> runeIndex;
364 }
365 );
366
367 if (not maybeBi)
368 return {li, 0, 0};
369
370 auto bi = *maybeBi;
371
372 auto& block = line.blocks()[bi];
373
374 auto maybeCi = searchLowerBound(
375 block.cells(), [&](Cell const& c) {
376 return c.runeRange.start <=> runeIndex;
377 }
378 );
379
380 if (not maybeCi)
381 return {li, bi, 0};
382
383 auto ci = *maybeCi;
384
385 auto cell = block.cells()[ci];
386
387 if (cell.runeRange.end() == runeIndex) {
388 // Handle the case where the rune is the last of the text
389 ci++;
390 }
391
392 return {li, bi, ci};
393 }
394
395 Vec2Au queryPosition(usize runeIndex) const {
396 auto [li, bi, ci] = lbcAt(runeIndex);
397
398 if (isEmpty(_lines))
399 return {};
400
401 auto& line = _lines[li];
402
403 if (isEmpty(line.blocks()))
404 return {0_au, line.baseline};
405
406 auto& block = line.blocks()[bi];
407
408 if (isEmpty(block.cells()))
409 return {block.pos, line.baseline};
410
411 if (ci >= block.cells().len()) {
412 // Handle the case where the rune is the last of the text
413 auto& cell = last(block.cells());
414 return {block.pos + cell.pos + cell.adv, cell.Yposition(line.baseline)};
415 }
416
417 auto& cell = block.cells()[ci];
418
419 return {block.pos + cell.pos, cell.Yposition(line.baseline)};
420 }
