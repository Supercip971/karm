210 Res<u8> readBit() {
211 if (_len == 0) {
212 try$(_reader.read(MutBytes{&_bits, 1}));
213 _len = 8;
214 }
215
216 u8 bit = _bits & 1;
217 _bits >>= 1;
218 _len -= 1;
