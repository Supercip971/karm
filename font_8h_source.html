367struct FontFamily : Fontface {
368 struct Member {
369 FontAdjust adjust;
370 Rc<Fontface> face;
371 Opt<Ranges<Range<Rune>>> ranges;
372 };
373
374 FontAdjust _adjust;
375 Vec<Member> _members;
376
377 FontFamily(Vec<Member> members) : _members(std::move(members)) {}
378
379 FontMetrics metrics() override {
380 FontMetrics metrics = {};
381
382 for (auto& member : _members) {
383 auto m = metrics.combine(member.face->metrics());
384 auto& a = member.adjust;
385
386 m.ascend *= a.ascent * _adjust.ascent;
387 m.descend *= a.descent * _adjust.descent;
388 m.linegap *= a.linegap * _adjust.linegap;
389
390 metrics = metrics.combine(m);
391 }
392
393 return metrics;
394 }
395
396 FontAttrs attrs() const override {
397 FontAttrs attrs;
398
399 StringBuilder familyName;
400 for (auto& member : _members) {
401 if (familyName.len())
402 familyName.append(" | "s);
403 familyName.append(member.face->attrs().family.str());
404 }
405
406 attrs.family = Symbol::from(familyName.take());
407
408 return attrs;
409 }
410
411 Glyph glyph(Rune rune) override {
412 Glyph res = Glyph::TOFU;
413
414 for (usize i = 0; i < _members.len(); i++) {
415 auto& member = _members[i];
416 if (member.ranges and not member.ranges->contains({rune, 1})) {
417 continue;
418 }
419
420 res = member.face->glyph(rune);
421 if (res != Glyph::TOFU) {
422 res.font = i;
423 break;
424 }
425 }
426
427 if (res == Glyph::TOFU)
428 logWarn("failed to find glyph for rune: {:x}", rune);
429
430 return res;
431 }
432
433 f64 advance(Glyph glyph) override {
434 auto& member = _members[glyph.font];
435 auto a = member.face->advance(glyph);
436 return a * member.adjust.sizeAdjust * _adjust.sizeAdjust;
437 }
438
439 f64 kern(Glyph prev, Glyph curr) override {
440 if (prev.font != curr.font)
441 return 0;
442
443 auto& member = _members[prev.font];
444 auto k = member.face->kern(prev, curr);
445 return k * member.adjust.sizeAdjust * _adjust.sizeAdjust;
446 }
447
448 void contour(Canvas& g, Glyph glyph) const override;
