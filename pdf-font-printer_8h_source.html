25 Pdf::Array fontBBox() {
26 // Values in the glyph coordinate system
27 f64 xMin = 0;
28 f64 xMax = 0;
29 f64 yMin = 0;
30 f64 yMax = 0;
31
32 for (auto [_, GID] : codeMappings.iter()) {
33 Text::Glyph glyph{.index = GID, .font = 0};
34 auto metrics = _font->_parser.glyphMetrics(glyph);
35
36 xMin = min(xMin, metrics.x - metrics.width);
37 xMax = max(xMax, metrics.x);
38 yMin = min(yMin, metrics.y - metrics.height);
39 yMax = max(yMax, metrics.y);
40 }
41
42 return Pdf::Array{xMin, yMin, xMax, yMax};
43 }

45 Pdf::Array widths() {
46 // 9.7.4.3 Glyph metrics in CIDFonts
47 // FIXME: (perf) when flushing, currGroupW can be empty only in the call outside of the loop
48 // FIXME: (perf) setting currGroupStart, happens either after flushing or in the first loop iteration
49
50 Pdf::Array allWidths;
51
52 u16 currGroupStart;
53 Pdf::Array currGroupW;
54
55 auto flushCollectedWidths = [&]() {
56 if (currGroupW.len() == 0)
57 return;
58 allWidths.pushBack(usize{currGroupStart});
59 allWidths.pushBack(std::move(currGroupW));
60 currGroupW = {};
61 };
62
63 Opt<u16> prevCid;
64 for (auto const& [cid, gid] : codeMappings.iter()) {
65
66 if (prevCid and prevCid.unwrap() + 1 != cid)
67 flushCollectedWidths();
68
69 if (currGroupW.len() == 0)
70 currGroupStart = cid;
71
72 Text::Glyph glyph{.index = gid, .font = 0};
73 // We are using the Em unit to relate font design units and width pdf units
74 // The unit in widths array is (Em/1000) (9.2.4 Glyph positioning and metrics)
75 auto gliyphAdvInEm = _font->advance(glyph);
76 currGroupW.pushBack(gliyphAdvInEm * 1000);
77
78 prevCid = cid;
79 }
80
81 flushCollectedWidths();
82
83 return allWidths;
84 }

86 Buf<u8> CIDToGIDMap() {
87 Buf<u8> buf;
88
89 u16 const BYTE_MASK = 255;
90
91 usize consecutiveUnmappedCIDs = 0;
92 for (usize i = 0; i < CODESPACE; ++i) {
93 auto glyph = codeMappings.tryGet(i);
94
95 if (not glyph) {
96 consecutiveUnmappedCIDs++;
97 continue;
98 }
99
100 for (usize j = 0; j < consecutiveUnmappedCIDs; ++j) {
101 buf.insert(buf.len(), 0);
102 buf.insert(buf.len(), 0);
103 }
104 consecutiveUnmappedCIDs = 0;
105
106 auto gid = glyph.unwrap();
107 buf.insert(buf.len(), (gid >> 8) & BYTE_MASK);
108 buf.insert(buf.len(), gid & BYTE_MASK);
109 }
110
111 return buf;
112 }
