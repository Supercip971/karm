315 try$(format(w, "{} ", cmd.longName));

321 for (auto& opt : options) {

389 Res<bool> _evalOption(Cursor<Token>& c) {
390 bool found = false;
391
392 for (auto& opt : options) {
393 if (c.ended())
394 break;
395
396 if (opt->kind != OptionKind::OPTION)
397 continue;
398
399 bool shortNameMatch = opt->shortName and c->flag == opt->shortName.unwrap();
400 bool longNameMatch = c->value == opt->longName;
401
402 if (not(shortNameMatch or longNameMatch))
403 continue;
404
405 c.next();
406 try$(opt->eval(c));
407 found = true;
408 }
409
410 return Ok(found);
411 }

413 Res<> _evalParams(Cursor<Token>& c) {

444 Async::Task<> execAsync(Sys::Context& ctx, Cursor<Token> c) {
445 co_try$(_evalParams(c));
446
447 if (_help) {
448 co_try$(_showHelp(Sys::out()));
449 co_return Ok();
450 }
451
452 if (_usage) {
453 co_try$(_showUsage(Sys::out()));
454 co_return Ok();
455 }
456
457 if (_version) {
458 co_try$(format(Sys::out(), "{} {}\n", longName, stringify$(__ck_version_value) ""s));
459 co_return Ok();
460 }
461
462 _invoked = true;
463 if (callbackAsync)
464 co_trya$(callbackAsync.unwrap()(ctx));
465
466 if (Karm::any(_commands) and c.ended()) {
467 co_try$(_showUsage(Sys::out()));
468 co_return Error::invalidInput("expected subcommand");
469 }
470
471 if (not c.ended()) {
472 if (c->kind != Token::OPERAND)
473 co_return Error::invalidInput("expected subcommand");
474
475 if (not Karm::any(_commands))
476 co_return Error::invalidInput("unexpected subcommand");
477
478 auto value = c->value;
479 c.next();
480
481 for (auto& cmd : _commands) {
482 if (value != cmd->longName)
483 continue;
484 co_return co_await cmd->execAsync(ctx, c);
485 }
486
487 co_return Error::invalidInput("unknown subcommand");
488 }
489 co_return Ok();
490 }
491
492 operator bool() const {
493 return _invoked;
