43 Res<Pair<ValueRecord>> adjustments(usize prev, usize curr) const {
44 // 1. Locate the current script in the GPOS ScriptList table.
45
46 // FIXME: We assume that the script is always "latn".
47 auto scriptTable = try$(scriptList().lookup("latn"));
48
49 // 2. If the language system is known, search the script for the correct
50 // LangSys table; otherwise, use the script’s default LangSys table.
51
52 // FIXME: We assume that the language system is always "dflt".
53 auto langSys = scriptTable.defaultLangSys();
54
55 // 3. The LangSys table provides index numbers into the GPOS FeatureList
56 // table to access a required feature and a number of additional features.
57 Opt<FeatureTable> kernFeatureTable;
58 for (auto featureIndex : langSys.iterFeatures()) {
59 auto featureTable = featureList().at(featureIndex);
60
61 // 4. Inspect the featureTag of each feature, and select the feature
62 // tables to apply to an input glyph string.
63 if (featureTable.tag == "kern") {
64 kernFeatureTable = featureTable;
65 break;
66 }
67 }
68
69 if (not kernFeatureTable)
70 return Ok(Pair<ValueRecord>{});
71
72 // 5. If a Feature Variation table is present, evaluate conditions in
73 // the Feature Variation table to determine if any of the initially-
74 // selected feature tables should be substituted by an alternate
75 // feature table.
76
77 // FIXME: We don't support feature variations.
78
79 // 6. Each feature provides an array of index numbers into the GPOS
80 // LookupList table. Assemble all lookups from the set of chosen
81 // feature tables, and apply the lookups in the order given in the
82 // LookupList table.
83 for (auto lookupIndex : kernFeatureTable->iterLookups()) {
84 auto lookupTable = lookupList().at(lookupIndex);
85
86 // FIXME: We only support pair adjustment lookups.
87 if (lookupTable.lookupType() != (u16)GposLookupType::PAIR_ADJUSTMENT)
88 continue;
89
90 for (auto lookupSubtable : lookupTable.iter()) {
91 if (auto glyphPair = lookupSubtable.is<GlyphPairAdjustment>()) {
92 auto pair = glyphPair->adjustments(prev, curr);
93 if (pair)
94 return Ok(*pair);
95 } else if (auto classPair = lookupSubtable.is<ClassPairAdjustment>()) {
96 auto pair = classPair->adjustments(prev, curr);
97 if (pair)
98 return Ok(*pair);
99 } else {
100 logWarn("ttf: unsupported GPOS lookup subtable");
101 }
102 }
103 }
104
105 return Ok(Pair<ValueRecord>{});
106 }
107};
