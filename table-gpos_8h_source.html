42 Res<Pair<ValueRecord>> adjustments(usize prev, usize curr) const {
43 // 1. Locate the current script in the GPOS ScriptList table.
44
45 // FIXME: We assume that the script is always "latn".
46 auto scriptTable = try$(scriptList().lookup("latn"));
47
48 // 2. If the language system is known, search the script for the correct
49 // LangSys table; otherwise, use the scriptâ€™s default LangSys table.
50
51 // FIXME: We assume that the language system is always "dflt".
52 auto langSys = scriptTable.defaultLangSys();
53
54 // 3. The LangSys table provides index numbers into the GPOS FeatureList
55 // table to access a required feature and a number of additional features.
56 Opt<FeatureTable> kernFeatureTable;
57 for (auto featureIndex : langSys.iterFeatures()) {
58 auto featureTable = featureList().at(featureIndex);
59
60 // 4. Inspect the featureTag of each feature, and select the feature
61 // tables to apply to an input glyph string.
62 if (featureTable.tag == "kern") {
63 kernFeatureTable = featureTable;
64 break;
65 }
66 }
67
68 if (not kernFeatureTable)
69 return Ok(Pair<ValueRecord>{});
70
71 // 5. If a Feature Variation table is present, evaluate conditions in
72 // the Feature Variation table to determine if any of the initially-
73 // selected feature tables should be substituted by an alternate
74 // feature table.
75
76 // FIXME: We don't support feature variations.
77
78 // 6. Each feature provides an array of index numbers into the GPOS
79 // LookupList table. Assemble all lookups from the set of chosen
80 // feature tables, and apply the lookups in the order given in the
81 // LookupList table.
82 for (auto lookupIndex : kernFeatureTable->iterLookups()) {
83 auto lookupTable = lookupList().at(lookupIndex);
84
85 // FIXME: We only support pair adjustment lookups.
86 if (lookupTable.lookupType() != (u16)GposLookupType::PAIR_ADJUSTMENT)
87 continue;
88
89 for (auto lookupSubtable : lookupTable.iter()) {
90 if (auto glyphPair = lookupSubtable.is<GlyphPairAdjustment>()) {
91 auto pair = glyphPair->adjustments(prev, curr);
92 if (pair)
93 return Ok(*pair);
94 } else if (auto classPair = lookupSubtable.is<ClassPairAdjustment>()) {
95 auto pair = classPair->adjustments(prev, curr);
96 if (pair)
97 return Ok(*pair);
98 } else {
99 logWarn("ttf: unsupported GPOS lookup subtable");
100 }
101 }
102 }
103
104 return Ok(Pair<ValueRecord>{});
105 }
