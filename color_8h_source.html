37 always_inline constexpr Color blendOver(Color const background) const {
38 if (alpha == 0xff) {
39 return *this;
40 } else if (alpha == 0) {
41 return background;
42 } else if (background.alpha == 255u) {
43 return {
44 static_cast<u8>((background.red * 255u * (255u - alpha) + 255u * alpha * red) / 65025),
45 static_cast<u8>((background.green * 255u * (255u - alpha) + 255u * alpha * green) / 65025),
46 static_cast<u8>((background.blue * 255u * (255u - alpha) + 255u * alpha * blue) / 65025),
47 static_cast<u8>(255),
48 };
49 } else {
50 u16 d = 255u * (background.alpha + alpha) - background.alpha * alpha;
51
52 return {
53 static_cast<u8>((background.red * background.alpha * (255u - alpha) + 255u * alpha * red) / d),
54 static_cast<u8>((background.green * background.alpha * (255u - alpha) + 255u * alpha * green) / d),
55 static_cast<u8>((background.blue * background.alpha * (255u - alpha) + 255u * alpha * blue) / d),

260static inline YCbCr rgbToYCbCr(Color color) {
261 YCbCr yCbCr;
262 yCbCr.y = 0.299 * color.red + 0.587 * color.green + 0.114 * color.blue;
263 yCbCr.cb = 128 - 0.168736 * color.red - 0.331264 * color.green + 0.5 * color.blue;
264 yCbCr.cr = 128 + 0.5 * color.red - 0.418688 * color.green - 0.081312 * color.blue;
265 return yCbCr;
266}

268static inline Color yCbCrToRgb(YCbCr yCbCr) {
269 float r = yCbCr.y + 1.402f * yCbCr.cr + 128;
270 float g = yCbCr.y - 0.344f * yCbCr.cb - 0.714f * yCbCr.cr + 128;
271 float b = yCbCr.y + 1.772f * yCbCr.cb + 128;
272
273 r = clamp(r, 0.0f, 255.0f);
274 g = clamp(g, 0.0f, 255.0f);
275 b = clamp(b, 0.0f, 255.0f);
276 return Gfx::Color::fromRgba(r, g, b, 255);
