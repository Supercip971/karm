84 constexpr bool straight(T epsilon = Limits<T>::EPSILON) const {
85 auto minx = min(ax, dy);
86 auto miny = min(ay, dy);
87 auto maxx = max(ax, dx);
88 auto maxy = max(ay, dy);
89
90 return
91 // Is P1 located between P0 and P3?
92 minx <= bx and
93 miny <= by and
94 maxx >= bx and
95 maxy >= by and
96 // Is P2 located between P0 and P3?
97 minx <= cx and
98 miny <= cy and
99 maxx >= cx and
100 maxy >= cy and
101 // Are all points collinear?
102 epsilonEq(Tri2(a, b, d).turn(), 0.0, epsilon) and
103 epsilonEq(Tri2(a, c, d).turn(), 0.0, epsilon);
104 }

106 constexpr bool straightish(T tolerance) const {
107 auto d1 = d - a;
108 auto d2 = Math::abs((b.x - d.x) * d1.y - (b.y - d.y) * d1.x);
109 auto d3 = Math::abs((c.x - d.x) * d1.y - (c.y - d.y) * d1.x);
110
111 return (d2 + d3) * (d2 + d3) < tolerance * (d1.x * d1.x + d1.y * d1.y);
112 }

114 constexpr Vec2<T> eval(T t) const {
115 auto u = 1 - t;
116 auto uu = u * u;
117 auto uuu = uu * u;
118 auto tt = t * t;
119 auto ttt = tt * t;
120
121 return a * uuu + b * 3 * uu * t + c * 3 * u * tt + d * ttt;
122 }

124 constexpr Vec2<T> derivative(T t) const {
125 auto u = 1 - t;
126 auto uu = u * u;
127 auto tt = t * t;
128
129 return (b - a) * 3 * uu + (c - b) * 6 * u * t + (d - c) * 3 * tt;
130 }

142 constexpr Curve offset(T offset) const {
143 Vec2<T> d1 = b - a;
144 Vec2<T> d2 = c - d;
145 auto div = d1.cross(d2);
146
147 if (epsilonEq(div, 0.0))
148 return offset2(offset);
149
150 // Start point.
151 Vec2<T> p0 = a + normal(0.0).unit() * offset;
152
153 // End point.
154 Vec2<T> p3 = d + normal(1.0).unit() * offset;
155
156 // Middle point.
157 Vec2<T> mp = eval(0.5);
158 Vec2<T> mpN = normal(0.5).unit();
159 Vec2<T> p = mp + (mpN * offset);
160
161 Vec2<T> bxby = (8.0 / 3.0) * (p - (0.5 * (p0 + p3)));
162
163 auto a = bxby.cross(d2) / div;
164 auto b = d1.cross(bxby) / div;
165
166 Vec2<T> p1{p0.x + a * d1.x, p0.y + a * d1.y};
167 Vec2<T> p2{p3.x + b * d2.x, p3.y + b * d2.y};
168
169 return {p0, p1, p2, p3};
170 }

172 constexpr Pair<Curve> split(T t) const {
173 auto e = lerp(a, b, t);
174 auto f = lerp(b, c, t);
175 auto g = lerp(c, d, t);
176 auto h = lerp(e, f, t);
177 auto j = lerp(f, g, t);
178 auto k = lerp(h, j, t);
179
180 return {
181 {a, e, h, k},
182 {k, j, g, d},
183 };
184 }

186 constexpr Pair<Curve> split() const {
187 auto ab = (a + b) / 2;
188 auto bc = (b + c) / 2;
189 auto cd = (c + d) / 2;
190 auto abc = (ab + bc) / 2;
191 auto bcd = (bc + cd) / 2;
192 auto abcd = (abc + bcd) / 2;
193
194 return {
195 {a, ab, abc, abcd},
196 {abcd, bcd, cd, d},
197 };

200 constexpr Curve offset2(T offset) const {
201 Curve res;
